// ============================================================ // ГЛОБАЛЬНАЯ ПЕРЕМЕННАЯ ДЛЯ БАЗЫ // ============================================================ let fullDatabase = []; // ============================================================ // ЗАГРУЗКА БАЗЫ (С ФИКСОМ ДЛЯ iOS) // ============================================================ async function loadDatabase() { try { // Принудительно добавляем параметр, чтобы избежать кэширования на iOS const url = 'csvjson.json?_=' + Date.now(); const response = await fetch(url, { cache: 'no-store', headers: { 'Cache-Control': 'no-cache' } }); if (!response.ok) throw new Error('HTTP ' + response.status); fullDatabase = await response.json(); if (!Array.isArray(fullDatabase) || fullDatabase.length === 0) { throw new Error('Пустая база'); } console.log('[+] База загружена:', fullDatabase.length, 'записей'); return true; } catch (e) { console.warn('Ошибка загрузки базы:', e); // РЕЗЕРВ: пробуем загрузить через raw.githubusercontent.com try { const fallbackUrl = 'https://raw.githubusercontent.com/ruslanakhmedov1337-netizen/bloodthirsty.github.io/main/csvjson.json?_=' + Date.now(); const resp2 = await fetch(fallbackUrl, { cache: 'no-store' }); if (resp2.ok) { fullDatabase = await resp2.json(); console.log('[+] База загружена через резерв:', fullDatabase.length, 'записей'); return true; } } catch (e2) { console.error('Резерв не сработал:', e2); } // Если всё упало — создаём тестовую базу fullDatabase = [ { name: 'Тест', phone: '79001234567', tg: '@test', email: 'test@mail.ru' }, { name: 'Иванов Иван', phone: '79161234567', tg: '@ivanov', email: 'ivan@mail.ru' } ]; console.warn('[!] Используется тестовая база (файл не найден)'); return false; } } // ============================================================ // ТВОЯ ФУНКЦИЯ (С МАЛЕНЬКИМ ФИКСОМ) // ============================================================ async function runTerminalSimulation() { // Проверяем, загружена ли база if (fullDatabase.length === 0) { await loadDatabase(); } document.getElementById('screen-loading').style.display = 'none'; document.getElementById('screen-terminal').style.display = 'block'; const input = document.getElementById('main-search-input').value.trim(); const output = document.getElementById('terminal-output'); // Если ввод пустой — показываем подсказку if (!input) { output.innerHTML = '[-] Введите данные для поиска.'; document.getElementById('btn-restart').style.display = 'block'; return; } // Очищаем ввод от всего, кроме цифр const cleanInput = input.replace(/\D/g, ''); // Ищем в базе: перебираем каждое поле каждого объекта const found = fullDatabase.find(item => { return Object.values(item).some(val => { const dbValue = String(val).replace(/\D/g, ''); // Сравниваем очищенные строки (если это цифры) if (cleanInput.length > 0 && dbValue.length > 0) { return dbValue.includes(cleanInput); } // Иначе ищем по точному текстовому совпадению return String(val).toLowerCase() === input.toLowerCase(); }); }); if (found) { let resultHtml = `[+] ДАННЫЕ НАЙДЕНЫ:
`; for (let key in found) { resultHtml += `• ${key}: ${found[key]}
`; } output.innerHTML = resultHtml; } else { output.innerHTML = `[-] Совпадений по запросу "${input}" не найдено.`; } document.getElementById('btn-restart').style.display = 'block'; } // ============================================================ // АВТОМАТИЧЕСКАЯ ЗАГРУЗКА БАЗЫ ПРИ СТАРТЕ // ============================================================ // Вызываем загрузку при загрузке страницы document.addEventListener('DOMContentLoaded', function() { loadDatabase().then(success => { console.log(success ? '✅ База готова' : '⚠️ База загружена с ошибками'); }); });
